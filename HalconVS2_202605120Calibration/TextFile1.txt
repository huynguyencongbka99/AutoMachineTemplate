6. Hàm Find Mark
private bool FindMark(
    HImage image,
    out double row,
    out double col)
{
    row = 0;
    col = 0;

    try
    {
        HTuple rows, cols, angles, scores;

        HOperatorSet.FindShapeModel(
            image,
            modelID,
            -0.39,
            0.78,
            0.5,
            1,
            0.5,
            "least_squares",
            0,
            0.8,
            out rows,
            out cols,
            out angles,
            out scores);

        if (scores.Length <= 0)
            return false;

        row = rows[0].D;
        col = cols[0].D;

        return true;
    }
    catch
    {
        return false;
    }
}

8. STEP 5 — Tính HomMat2D

Đây là phần quan trọng nhất.

Hàm Calculate Calibration
public bool CalculateCalibration()
{
    try
    {
        if (calibPoints.Count < 4)
            return false;

        HTuple imageX = new HTuple();
        HTuple imageY = new HTuple();

        HTuple robotX = new HTuple();
        HTuple robotY = new HTuple();

        foreach (var p in calibPoints)
        {
            imageX = imageX.TupleConcat(p.ImageX);
            imageY = imageY.TupleConcat(p.ImageY);

            robotX = robotX.TupleConcat(p.RobotX);
            robotY = robotY.TupleConcat(p.RobotY);
        }

        HOperatorSet.VectorToHomMat2d(
            imageX,
            imageY,
            robotX,
            robotY,
            out homMat2D);

        return true;
    }
    catch
    {
        return false;
    }
}

Hàm Convert Image → Robot
public bool ImageToRobot(
    double imageRow,
    double imageCol,
    out double robotX,
    out double robotY)
{
    robotX = 0;
    robotY = 0;

    try
    {
        HTuple x, y;

        HOperatorSet.AffineTransPoint2d(
            homMat2D,
            imageCol,
            imageRow,
            out x,
            out y);

        robotX = x.D;
        robotY = y.D;

        return true;
    }
    catch
    {
        return false;
    }
}
